200template <class T>
201bool Vertex<T>::removeEdge(T in) {
202 bool removedEdge = false;
203 auto it = adj.begin();
204 while (it != adj.end()) {
205 Edge<T> *edge = *it;
206 Vertex<T> *dest = edge->getDest();
207 if (dest->getInfo() == in) {
208 it = adj.erase(it);
209 deleteEdge(edge);
210 removedEdge = true; // allows for multiple edges to connect the same pair of vertices (multigraph)
211 }
212 else {
213 it++;
214 }
215 }
216 return removedEdge;
217}

404template <class T>
405void Vertex<T>::deleteEdge(Edge<T> *edge) {
406 Vertex<T> *dest = edge->getDest();
407 // Remove the corresponding edge from the incoming list
408 auto it = dest->incoming.begin();
409 while (it != dest->incoming.end()) {
410 if ((*it)->getOrig()->getInfo() == info) {
411 it = dest->incoming.erase(it);
412 }
413 else {
414 it++;
415 }
416 }
417 delete edge;
418}

743/****************** BFS ********************/
751template <class T>
752std::vector<T> Graph<T>::bfs(const T & source) const {
753 std::vector<int> res;
754 // Get the source vertex
755 auto s = findVertex(source);
756 if (s == nullptr) {
757 return res;
758 }
759
760 // Set that no vertex has been visited yet
761 for (auto v : vertexSet) {
762 v->setVisited(false);
763 }
764
765 // Perform the actual BFS using a queue
766 std::queue<Vertex<T> *> q;
767 q.push(s);
768 s->setVisited(true);
769 while (!q.empty()) {
770 auto v = q.front();
771 q.pop();
772 res.push_back(v->getInfo());
773 for (auto & e : v->getAdj()) {
774 auto w = e->getDest();
775 if ( ! w->isVisited()) {
776 q.push(w);
777 w->setVisited(true);
778 }
779 }
780 }
781 return res;
782}

814template <class T>
815bool Graph<T>::dfsIsDAG(Vertex<T> *v) const {
816 v->setVisited(true);
817 v->setProcesssing(true);
818 for (auto e : v->getAdj()) {
819 auto w = e->getDest();
820 if (w->isProcessing()) return false;
821 if (! w->isVisited()) {
822 if (! dfsIsDAG(w)) return false;
823 }
824 }
825 v->setProcesssing(false);
826 return true;

841std::vector<T> Graph<T>::topsort() const {
842 std::vector<int> res;
843
844 for (auto v : vertexSet) {
845 v->setIndegree(0);
846 }
847 for (auto v : vertexSet) {
848 for (auto e : v->getAdj()) {
849 unsigned int indegree = e->getDest()->getIndegree();
850 e->getDest()->setIndegree(indegree + 1);
851 }
852 }
853
854 std::queue<Vertex<T> *> q;
855 for (auto v : vertexSet) {
856 if (v->getIndegree() == 0) {
857 q.push(v);
858 }
859 }
860
861 while( !q.empty() ) {
862 Vertex<T> * v = q.front();
863 q.pop();
864 res.push_back(v->getInfo());
865 for(auto e : v->getAdj()) {
866 auto w = e->getDest();
867 w->setIndegree(w->getIndegree() - 1);
868 if(w->getIndegree() == 0) {
869 q.push(w);
870 }
871 }
872 }
873
874 if ( res.size() != vertexSet.size() ) {
875 //std::cout << "Impossible topological ordering!" << std::endl;
876 res.clear();
877 return res;
878 }
879
880 return res;
881}
